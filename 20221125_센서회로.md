#include <avr/io.h>
#include <avr/interrupt.h>
#include <util/delay.h>

void initport(){
	DDRB=DDRB|0b00100000;
	
}

void inittimer(){
	TCCR1A=0x82;
	TCCR1B=0x1a;
	
	ICR1=40000;
	OCR1A=1200;
	
}

void setAg(int i){
	
	if (OCR1A > 5000)
	{
		OCR1A=1200;
	}
	
	OCR1A+=i;
}

int main(void)
{
	inittimer();
	initport();
	
	sei();
	
    /* Replace with your application code */
    while (1) 
    {
		
		setAg(10);
		
		_delay_ms(1000);
    }
}
